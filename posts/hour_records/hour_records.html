hour_records |>
  ggplot() +
  aes(x = date, y = distance, colour = type, fill = type) +
  geom_point() +
  geom_line() +
  scale_y_continuous(limits = c(25, 60), breaks = seq(25, 60, 5)) +
  scale_x_date(limits = c(as.Date("1890-01-01"), NA), 
               breaks = as.Date(paste0(seq(1890, 2030, by = 10), "-01-01")), 
               labels = date_format("%Y")) +
  scale_colour_viridis(discrete = TRUE) +
  scale_fill_viridis(discrete = TRUE) +
  facet_grid(rows = vars(gender)) +
  labs(title = "Cycling Hour Records",
       subtitle = "1890-present",
       y = "Distance (km)", 
       x = "Date", 
       colour = "", 
       fill = "") +
  theme_jdonland() +
  theme(legend.position = "top")
